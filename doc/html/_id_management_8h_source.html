284  n = std::abs((n / static_cast<ldouble>(step)) + 1.0);

320  template<class T>
321  T findNearestStdId(T target, T start, T step) {
322  ldouble An = static_cast<ldouble>(target);
323  ldouble A1 = static_cast<ldouble>(start);
324  ldouble d = static_cast<ldouble>(target > start ? ONF::abs(step) : -ONF::abs(step));
325 
326  ldouble n = ((An - A1) / d) + 1.0;
327 
328  if (n - static_cast<udlong>(n) == 0.0)
329  return target;
330 
331  return A1 + (std::floor(n) - 1.0) * d;
332  }
