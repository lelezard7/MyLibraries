287 bool isStandardId(T target, T start, T step) {
288 ldouble n = static_cast<ldouble>(target) - static_cast<ldouble>(start);
289 n = std::abs((n / static_cast<ldouble>(step)) + 1.0);
290
291 return (n - static_cast<udlong>(n) == 0.0);
292 }

323 template<class T>
324 T findNearestStdId(T target, T start, T step) {
325 ldouble An = static_cast<ldouble>(target);
326 ldouble A1 = static_cast<ldouble>(start);
327 ldouble d = static_cast<ldouble>(target > start ? ONF::abs(step) : -ONF::abs(step));
328
329 ldouble n = ((An - A1) / d) + 1.0;
330
331 if (n - static_cast<udlong>(n) == 0.0)
332 return target;
333
334 return A1 + (std::floor(n) - 1.0) * d;
335 }
