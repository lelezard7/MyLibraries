222  bool isStandardId(T target, T start, T step) {
223  ldouble An = static_cast<ldouble>(target);
224  ldouble A1 = static_cast<ldouble>(start);
225  ldouble d = static_cast<ldouble>(step);
226 
227  ldouble n = std::abs((An - A1 + d) / d);
228  return (n - static_cast<udlong>(n) == static_cast<ldouble>(0.0));
229  }
