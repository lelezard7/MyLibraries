261 n = std::abs((n / static_cast<ldouble>(step)) + 1.0);

295 template<class T>
296 T findNearestStdId(T target, T start, T step) {
297 ldouble An = static_cast<ldouble>(target);
298 ldouble A1 = static_cast<ldouble>(start);
299 ldouble d = static_cast<ldouble>(target > start ? ONF::abs(step) : -ONF::abs(step));
300
301 ldouble n = ((An - A1) / d) + 1.0;
302
303 if (n - static_cast<udlong>(n) == 0.0)
304 return target;
305
306 return A1 + (std::floor(n) - 1.0) * d;
307 }
